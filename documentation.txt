1. Project Overview

1.1 Architecture Summary
This project implements a hybrid mobile application architecture where native Android and iOS
applications embed a Flutter module for core functionality. The architecture leverages the strengths
of both native development (platform-specific features, performance optimization) and cross-
platform Flutter development (shared UI components, rapid iteration).

The native layer handles authentication flow, session management, and platform integrations, while
the Flutter module manages the product listing, detail views, and payment initiation flow.
Communication between layers occurs through MethodChannel on both platforms, enabling
bidirectional data exchange.


1.2 Technology Stack
The following table summarizes the core technologies used across all three components of the
application:


           Component                                                Technology

  Android Native                   Kotlin, SDK 35, Jetpack Compose/XML

  iOS Native                       Swift 5.9+, SwiftUI/UIKit

  Flutter Module                   Flutter 3.16+, Dart 3.2+

  State Management                 Riverpod (recommended) / Bloc / Provider

  Network Layer                    Dio with interceptors for logging

  API Source                       FakeStoreAPI (https://fakestoreapi.com)
                                    Table 1: Technology Stack Overview


1.3 Application Flow
The application follows a sequential flow that ensures seamless integration between native and
Flutter layers while maintaining state consistency across platform boundaries:

Launch App → Check UUID Cache → (If no UUID) Show Login Screen → Generate UUID → Pass
Configuration to Flutter → Load Flutter Module → Display Product List → Navigate to Detail →
Initiate Payment → Return Payload to Native.

    class NativeChannelService {
      static const MethodChannel _channel =
          MethodChannel('com.app.native/flutter');


        Future<AppConfig> getAppConfig() async {
          try {
            final Map<dynamic, dynamic> config =
                await _channel.invokeMethod('getAppConfig');
            return AppConfig.fromJson(Map<String, dynamic>.from(config));
          } on PlatformException catch (e) {
            throw Exception('Failed to get app config: ${e.message}');
          }
        }


        Future<void> initiatePayment(Map<String, dynamic> payload) async {
          try {
            await _channel.invokeMethod('initiatePayment', payload);
          } on PlatformException catch (e) {
            throw Exception('Failed to initiate payment: ${e.message}');
          }
        }
    }

   import android.content.Context
   import android.content.Intent
   import android.os.Bundle
   import android.widget.Button

   import androidx.appcompat.app.AppCompatActivity
   import java.util.UUID


   class LoginActivity : AppCompatActivity() {
       private lateinit var etUsername: EditText
       private lateinit var etPassword: EditText
       private lateinit var btnLogin: Button


       override fun onCreate(savedInstanceState: Bundle?) {
           super.onCreate(savedInstanceState)
           setContentView(R.layout.activity_login)

            etUsername = findViewById(R.id.etUsername)
            etPassword = findViewById(R.id.etPassword)
            btnLogin = findViewById(R.id.btnLogin)

            btnLogin.setOnClickListener {
                if (validateInput()) performLogin()
            }
       }

       private fun validateInput(): Boolean {
           if (etUsername.text.isEmpty()) {
               etUsername.error = "Username required"
               return false
           }
           if (etPassword.text.length < 4) {
               etPassword.error = "Password must be 4+ chars"
               return false
           }
           return true
       }

       private fun performLogin() {
           val uuid = UUID.randomUUID().toString()
           getSharedPreferences("AppPrefs", MODE_PRIVATE)
               .edit().putString("uuid", uuid).apply()
           startActivity(Intent(this, MainActivity::class.java))
           finish()
       }
   }

4.3 Implement Flutter Coordinator
Create a coordinator that manages the Flutter engine and handles MethodChannel communication.

FlutterCoordinator.swift
   import SwiftUI
   import Flutter


   class FlutterCoordinator: ObservableObject {
       static let shared = FlutterCoordinator()
       var flutterEngine: FlutterEngine?
       var methodChannel: FlutterMethodChannel?

        func initialize() {
            flutterEngine = FlutterEngine(name: "io.flutter")
            flutterEngine?.run()

            methodChannel = FlutterMethodChannel(
                name: "com.app.native/flutter",
                binaryMessenger: flutterEngine!.binaryMessenger
            )

            methodChannel?.setMethodCallHandler { [weak self] call, result in
                switch call.method {
                case "getAppConfig":
                    result(self?.buildAppConfig())
                case "initiatePayment":
                    if let payload = call.arguments as? [String: Any] {
                         self?.handlePaymentRequest(payload: payload)
                    }
                    result(nil)
                default:
                    result(FlutterMethodNotImplemented)
                }
            }
        }

        private func buildAppConfig() -> [String: Any] {
            let uuid = UserDefaults.standard.string(forKey: "uuid") ?? ""
            return [
                "uuid": uuid,
                "themeMode": "light",
                "colors": ["primary": "#2196F3", "background": "#FFFFFF"],
                "spacing": ["small": 8.0, "medium": 16.0, "large": 24.0]
            ]
        }
